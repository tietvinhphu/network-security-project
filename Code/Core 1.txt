Cấu hình Core 1
------------------
1/Cấu hình cơ bản

enable
configure terminal
banner motd ^
+===========================+
|  FPT JETKING - PENTAGUARD |
+===========================+
^
hostname CORE1
no cdp run
no ip domain-lookup
line console 0
logging synchronous
end
copy run start

#--- Tạo vlan quản trị

enable
configure terminal
vlan 100
name MGMT
interface range e3/2-3
sw mode acc
sw acc vlan 100
int vlan 100
ip address 192.168.100.200 255.255.255.0
no shutdown
end
copy run start

3/Cấu hình EtherChannel

enable
configure terminal

interface range e0/2-3
description UPLINK to Core2 [Po1]
no switchport
shutdown
channel-group 1 mode active
no shutdown
exit
interface port-channel 1
no switchport
ip address 10.1.1.1 255.255.255.252
no shutdown
exit

interface range e1/0-1
description UPLINK to SW1 [Po2]
shutdown
channel-group 2 mode active
no shutdown
interface port-channel 2
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allow vlan all
exit

interface range e1/2-3
description UPLINK to SW2 [Po3]
shutdown
channel-group 3 mode active
no shutdown
interface port-channel 3
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allow vlan all
exit

end
copy run start
show etherchannel summary

12/Cấu hình SSH &  tạo ACL cho phép MGMT - SSH

enable
configure terminal
service password-encryption
enable secret @Penta123
ip domain-name pentaguard.local
username admin privilege 15 secret @Penta123
crypto key generate rsa
1024
ip ssh version 2

ip access-list standard VTY_SSH_FROM_MGMT_ONLY
permit host 192.168.100.8 log
deny any log
exit
line vty 0 4
access-class VTY_SSH_FROM_MGMT_ONLY in
transport input ssh
login local
end
copy run start

13/Tạo ACLs cho VLANs
#--- Tạo ACL cho phép VLAN90 quyền quản trị 

ena
conf t
ip access-list extended ACL_FROM_VLAN90
permit ip 192.168.90.0 0.0.0.255 any
exit
interface Vlan90
ip access-group ACL_FROM_VLAN90 in
exit

#--- Tạo ACL cấm các VLAN User giao tiếp với nhau

permit tcp any 192.168.90.0 0.0.0.255 established
deny ip any 192.168.0.0 0.0.255.255 log
permit ip any any
